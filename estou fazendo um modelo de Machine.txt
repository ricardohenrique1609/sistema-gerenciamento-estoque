

SELECT DISTINCT 
    descricao_polo_administrativo,
    descricao_predio_administrativo
FROM 
    db_corp_rh_analyticsapreports_spec_01.hub_people_analytics_colaborador
WHERE 
    texto_cidade_local_departamento = 'SAO PAULO'
    AND (
        -- Filtro genérico para tentar pegar os complexos que você citou
        descricao_polo_administrativo LIKE '%FARIA LIMA%' OR
        descricao_polo_administrativo LIKE '%ACO%' OR
        descricao_polo_administrativo LIKE '%CONCEICAO%' OR
        descricao_polo_administrativo LIKE '%JABAQUARA%' OR
        descricao_polo_administrativo LIKE '%ALFREDO%' OR
        descricao_polo_administrativo LIKE '%WALTER%' OR
        descricao_polo_administrativo LIKE '%OLAVO%' OR
        descricao_polo_administrativo LIKE '%EUDORO%' OR
        descricao_polo_administrativo LIKE '%TECNOLOGICO%' OR
        descricao_polo_administrativo LIKE '%CT%'
    )

    descricao_polo_administrativo;
